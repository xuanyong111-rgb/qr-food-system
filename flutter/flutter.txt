// === 文件名：浏览器控制台 (批量生成 Hash) ===
async function getMultipleHashes(pwdArray) {
const results = {};
for (const pwd of pwdArray) {
const buf = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(pwd));
results[pwd] = Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2, '0')).join('');
}
console.log(JSON.stringify(results, null, 2));
}
